    <script>
        // Generate calendar days
        function generateCalendar() {
            const calendarEl = document.querySelector('.calendar');
            // Clear any existing days (except header)
            while (calendarEl.childNodes.length > 7) {
                calendarEl.removeChild(calendarEl.lastChild);
            }
            
            // Generate days (for demo, we'll use June 2023)
            const daysInMonth = 30;
            const startDay = 4; // June 1, 2023 is Thursday
            
            // Add empty days for start of month
            for (let i = 0; i < startDay; i++) {
                const emptyDay = document.createElement('div');
                emptyDay.className = 'calendar-day';
                calendarEl.appendChild(emptyDay);
            }
            
            // Add actual days
            for (let i = 1; i <= daysInMonth; i++) {
                const day = document.createElement('div');
                day.className = 'calendar-day';
                if (i === 15) day.classList.add('active');
                if (i % 5 === 0 || i % 7 === 0) day.classList.add('has-events');
                day.textContent = i;
                calendarEl.appendChild(day);
            }
        }
        
        // Toggle chat window
        document.querySelector('.lilo-assistant').addEventListener('click', function() {
            document.querySelector('.lilo-chat-window').classList.toggle('open');
        });
        
        // Task completion toggle
        document.querySelectorAll('.task-checkbox').forEach(checkbox => {
            checkbox.addEventListener('click', function() {
                this.classList.toggle('checked');
            });
        });
        
        // Alarm statistics chart
        function initChart() {
            const ctx = document.getElementById('alarmChart').getContext('2d');
            new Chart(ctx, {
                type: 'line',
                data: {
                    labels: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'],
                    datasets: [{
                        label: 'Alarm Completions',
                        data: [12, 19, 14, 15, 16, 12, 10],
                        borderColor: '#0B6EFD',
                        tension: 0.3,
                        fill: true,
                        backgroundColor: 'rgba(11, 110, 253, 0.1)'
                    }]
                },
                options: {
                    plugins: {
                        legend: {
                            display: false
                        }
                    },
                    scales: {
                        y: {
                            beginAtZero: true,
                            grid: {
                                drawBorder: false
                            }
                        },
                        x: {
                            grid: {
                                display: false
                            }
                        }
                    }
                }
            });
        }
        
        // Initialize
        document.addEventListener('DOMContentLoaded', function() {
            generateCalendar();
            initChart();
            
            // Add click handler for quick add button
            document.querySelector('.quick-add-btn').addEventListener('click', function() {
                document.querySelector('.lilo-chat-window').classList.add('open');
            });
        });
